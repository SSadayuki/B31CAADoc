  CATGSMUnfoldTargetOrientationMode_ReverseUReverseV,

  CATGSMUnfoldTargetOrientationMode_SwapReverseV,
  CATGSMUnfoldTargetOrientationMode_SwapReverseUReverseV,

no axis inversion
